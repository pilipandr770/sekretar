Integration Example

To integrate the widget into your website, include the following code:

<!-- Include Socket.IO library --> <script src="https://cdn.socket.io/4.7.2/socket.io.min.js"></script> <!-- Include AI Secretary Widget --> <script src="/static/js/ai-secretary-widget.js"></script> <script> // Initialize widget with your configuration const widget = new AISecretaryWidget({ apiUrl: 'https://your-api-domain.com', socketUrl: 'https://your-api-domain.com', tenantId: 'your-tenant-id', apiKey: 'your-api-key', theme: 'light', position: 'bottom-right', title: 'AI Secretary', subtitle: 'How can I help you today?', autoOpen: false, enableSound: true, onMessage: function(message) { console.log('New message:', message); }, onConnect: function() { console.log('Widget connected'); } }); </script>

Configuration Options

{ // Required tenantId: 'your-tenant-id', apiKey: 'your-api-key', // API Configuration apiUrl: 'http://localhost:5000', socketUrl: 'http://localhost:5000', // Appearance theme: 'light', // 'light' or 'dark' position: 'bottom-right', // 'bottom-right', 'bottom-left', 'top-right', 'top-left' width: 350, height: 500, // Content title: 'AI Secretary', subtitle: 'How can I help you today?', placeholder: '{{ _("Type") }} your message...', // Behavior autoOpen: false, showTypingIndicator: true, showTimestamps: true, enableSound: true, // Styling customCSS: null, // Callbacks onMessage: function(message) { /* handle new message */ }, onConnect: function() { /* handle connection */ }, onDisconnect: function() { /* handle disconnection */ }, onError: function(error) { /* handle error */ } }
